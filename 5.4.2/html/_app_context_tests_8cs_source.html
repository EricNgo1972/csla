62  for (int x = 0; x < 10; x++)
63  {
64  AppContextThread act = new AppContextThread("Thread: " + x);
65  AppContextThreadList.Add(act);
66 
67  Thread t = new Thread(new ThreadStart(act.Run));
68  t.Name = "Thread: " + x;
69  t.Start();
70  ThreadList.Add(t);
71  }
72 
73  ApplicationContext.Clear();
74  Exception ex = null;
75  try
76  {
77  foreach (AppContextThread act in AppContextThreadList)
78  {
79  //We are accessing the Client/GlobalContext via this thread, therefore
80  //it should be removed.
81  Assert.AreEqual(true, act.Removed);
82  }
83  //We are now accessing the shared value. If any other thread
84  //loses its Client/GlobalContext this will turn to true
85  //Assert.AreEqual(false, AppContextThread.StaticRemoved);
86  }
87  catch (Exception e)
88  {
89  ex = e;
90  }
91  finally
92  {
93  foreach (AppContextThread act in AppContextThreadList)
94  act.Stop();
95 
96  foreach (Thread t in ThreadList)
97  {
98  t.Join();
99  }
100  }
101  if (ex != null) throw ex;
102  }
