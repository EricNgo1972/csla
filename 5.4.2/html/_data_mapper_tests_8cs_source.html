198

202  source.MyDouble = 456;
203  source.MyBool = true;
204  source.MyEnum = DataMapEnum.Second;
205  var g = Guid.NewGuid();
206  source.MyGuid = g;
207  source.MyNInt = 321;
208  source.MySmartDate = new Csla.SmartDate(new DateTime(2002, 12, 4));
209  source.MyDate = new DateTime(2002, 11, 2);
210  source.MyString = "Third";
211 
212  map.AddFieldMapping("_int", "_int");
213  map.AddFieldToPropertyMapping("_double", "MyDouble");
214  map.AddPropertyMapping("MyBool", "MyBool");
215  map.AddPropertyToFieldMapping("MyGuid", "_guid");
216  map.AddPropertyMapping("MyEnum", "MyString");
217  map.AddPropertyMapping("MyString", "MyEnum");
218 
219  Csla.Data.DataMapper.Map(source, target, map);
220 
221  Assert.AreEqual(123, target.MyInt, "Int should match");
222  Assert.AreEqual(456, target.MyDouble, "Double should match");
223  Assert.AreEqual(true, target.MyBool, "bool should match");
224  Assert.AreEqual(g, target.MyGuid, "guid should match");
225  Assert.AreEqual("Second", target.MyString, "string should match (converted enum)");
226  Assert.AreEqual(DataMapEnum.Third, target.MyEnum, "enum should match (parsed enum)");
227  Assert.AreNotEqual(source.MyDate, target.MyDate, "Dates should not match");
