29  public void AscendingSort()
30  {

32  SortedBindingList<int> sortedList = new SortedBindingList<int>(intArr);
33  sortedList.ListChanged += new ListChangedEventHandler(sortedList_ListChanged);
34 
35  Assert.AreEqual(false, sortedList.IsSorted);
36  Assert.AreEqual(56, intArr[3]);
37  sortedList.ApplySort("", ListSortDirection.Ascending);
38  Assert.AreEqual(44, sortedList[2]);
39  Assert.AreEqual(8, sortedList.Count);
40  Assert.AreEqual(true, sortedList.Contains(56));
41  Assert.AreEqual(4, sortedList.IndexOf(56));
42  Assert.AreEqual(true, sortedList.IsReadOnly);
43  Assert.AreEqual(true, sortedList.IsSorted);
44 
45  foreach (int item in sortedList)
46  {
47  Console.WriteLine(item.ToString());
48  }
49 
50  sortedList.RemoveSort();
51  Assert.AreEqual(false, sortedList.IsSorted);
52  Assert.AreEqual(56, sortedList[3]);
53 
54  // This list dowes not support searching
55  Assert.IsFalse(sortedList.SupportsSearching);
