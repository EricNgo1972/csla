103  var root = await Csla.DataPortal.CreateAsync<HasRulesManager>(new HasRulesManager.Criteria());

105  context.Assert.AreEqual(true, root.IsValid, "should be valid on create");
106  context.Assert.AreEqual(0, root.BrokenRulesCollection.Count);
107 
108  bool validationComplete = false;
109  root.ValidationComplete += (vo, ve) => { validationComplete = true; };
110 
111  root.BeginEdit();
112  root.Name = "";
113  context.Assert.AreEqual("", root.Name);
114  context.Assert.AreEqual(false, root.IsValid);
115  context.Assert.AreEqual(1, root.BrokenRulesCollection.Count);
116  context.Assert.AreEqual("Name required", root.BrokenRulesCollection[0].Description);
117  context.Assert.IsTrue(validationComplete, "ValidationComplete should have run");
118  root.BeginEdit();
119  root.Name = "Begin 1";
120  context.Assert.AreEqual("Begin 1", root.Name);
121  context.Assert.AreEqual(true, root.IsValid);
122  context.Assert.AreEqual(0, root.BrokenRulesCollection.Count);
123  root.BeginEdit();
124  root.Name = "Begin 2";
125  context.Assert.AreEqual("Begin 2", root.Name);
126  context.Assert.AreEqual(true, root.IsValid);
127  context.Assert.AreEqual(0, root.BrokenRulesCollection.Count);
128  root.BeginEdit();
129  root.Name = "Begin 3";
130  context.Assert.AreEqual("Begin 3", root.Name);
131  context.Assert.AreEqual(true, root.IsValid);
132  context.Assert.AreEqual(0, root.BrokenRulesCollection.Count);
133 
134  HasRulesManager hrmClone = root.Clone();
135 
136  //Test validation rule cancels for both clone and cloned
137  root.CancelEdit();
138  context.Assert.AreEqual("Begin 2", root.Name);
139  context.Assert.AreEqual(true, root.IsValid);
140  context.Assert.AreEqual(0, root.BrokenRulesCollection.Count);
141  hrmClone.CancelEdit();
142  context.Assert.AreEqual("Begin 2", hrmClone.Name);
143  context.Assert.AreEqual(true, hrmClone.IsValid);
144  context.Assert.AreEqual(0, hrmClone.BrokenRulesCollection.Count);
145  root.CancelEdit();
146  context.Assert.AreEqual("Begin 1", root.Name);
147  context.Assert.AreEqual(true, root.IsValid);
148  context.Assert.AreEqual(0, root.BrokenRulesCollection.Count);
149  hrmClone.CancelEdit();
150  context.Assert.AreEqual("Begin 1", hrmClone.Name);
151  context.Assert.AreEqual(true, hrmClone.IsValid);
152  context.Assert.AreEqual(0, hrmClone.BrokenRulesCollection.Count);
153  root.CancelEdit();
154  context.Assert.AreEqual("", root.Name);
155  context.Assert.AreEqual(false, root.IsValid);
156  context.Assert.AreEqual(1, root.BrokenRulesCollection.Count);
157  context.Assert.AreEqual("Name required", root.BrokenRulesCollection[0].Description);
158  hrmClone.CancelEdit();
159  context.Assert.AreEqual("", hrmClone.Name);
160  context.Assert.AreEqual(false, hrmClone.IsValid);
161  context.Assert.AreEqual(1, hrmClone.BrokenRulesCollection.Count);
162  context.Assert.AreEqual("Name required", hrmClone.BrokenRulesCollection[0].Description);
163  root.CancelEdit();
164  context.Assert.AreEqual("<new>", root.Name);
165  context.Assert.AreEqual(true, root.IsValid);
166  context.Assert.AreEqual(0, root.BrokenRulesCollection.Count);
167  hrmClone.CancelEdit();
168  context.Assert.AreEqual("<new>", hrmClone.Name);
169  context.Assert.AreEqual(true, hrmClone.IsValid);
170  context.Assert.AreEqual(0, hrmClone.BrokenRulesCollection.Count);

173  context.Complete();
